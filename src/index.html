const dates = narratives.map(d => new Date(d.P_Date));
const minDate = new Date(Math.min(...dates));
const maxDate = new Date(Math.max(...dates));
const endDate = Inputs.date({ value: new Date(maxDate), label: "აარჩიეთ საბოლოო თარიღი" });
const startDate = Inputs.date({ value: new Date(maxDate.setMonth(maxDate.getMonth() - 1)), label: "აარჩიეთ საწყისი თარიღი" });
function createNarrativeChart(data, title) {
  const aggregatedTop = Object.entries(
    data.reduce((acc, { narrative_text, n }) => {
      acc[narrative_text] = (acc[narrative_text] || 0) + n;
      return acc;
    }, {})
  )
  .sort(([, a], [, b]) => b - a)
  .slice(0, 7);

  return Plot.plot({
    width: 700,
    height: 400,
    marginLeft: 150,
    marks: [
      Plot.barX(aggregatedTop, { x: ([,count]) => count, y: ([text]) => text, sort: {y: "x", reverse: true}, tip: true }),
      Plot.ruleX([0])
    ],
    x: {label: "Frequency"},
    y: {label: null, tickFormat: d => d.length > 10 ? d.slice(0, 10) + '...' : d},
    title,
  });
}
const monitoringGroups = [...new Set(narratives.map(d => d.monitoring_group))];

const tabs = Inputs.tab(monitoringGroups, {label: "Monitoring Group"});
function updateFilteredChart() {
  const filteredData = narratives.filter(d =>
    d.P_Date >= startDate.value &&
    d.P_Date <= endDate.value &&
    d.monitoring_group === tabs.value
  );

  const chart = createNarrativeChart(filteredData, `Top Narratives for ${tabs.value}`);

  document.getElementById("filteredNarrativesChartDiv").innerHTML = "";
  document.getElementById("filteredNarrativesChartDiv").appendChild(chart);
}

startDate.addEventListener("input", updateFilteredChart);
endDate.addEventListener("input", updateFilteredChart);
tabs.addEventListener("input", updateFilteredChart);

Promise.all([dailyPosts, narratives]).then(() => {
  updateFilteredChart();
});
